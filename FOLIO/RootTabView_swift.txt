import SwiftUI

struct RootTabView: View {
    @Environment(PortfolioStore.self) private var store

    var body: some View {
        TabView {
            DashboardView()
                .tabItem { Label("Dashboard", systemImage: "chart.pie.fill") }
            PortfolioView()
                .tabItem { Label("Portfolio", systemImage: "square.stack.3d.up.fill") }
            WishlistView()
                .tabItem {
                    Label("Wishlist", systemImage: "star.fill")
                }
                .badge(store.triggeredAlerts.isEmpty ? 0 : store.triggeredAlerts.count)
            SoldView()
                .tabItem { Label("Ledger", systemImage: "list.bullet.rectangle.fill") }
        }
        .tint(Theme.accent)
    }
}
