//
//  AddCardForm.swift
//  Folio
//
//  Step two of adding an owned card: condition, purchase price, quantity, date.
//

import SwiftUI

struct AddCardForm: View {
    let card: Card
    @Environment(PortfolioStore.self) private var store
    @Environment(\.dismiss) private var dismiss

    @State private var condition: CardCondition = .nearMint
    @State private var pricePaidText = ""
    @State private var quantity = 1
    @State private var purchaseDate = Date()

    private var pricePaid: Double { Double(pricePaidText) ?? 0 }
    private var adjustedValue: Double { card.marketPrice * condition.priceMultiplier }

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.background.ignoresSafeArea()
                ScrollView {
                    VStack(spacing: 20) {
                        header
                        conditionSection
                        purchaseSection
                        valuePreview
                    }
                    .padding(16)
                }
            }
            .navigationTitle("Add to Portfolio")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .topBarLeading) {
                    Button("Cancel") { dismiss() }.tint(Theme.textSecondary)
                }
                ToolbarItem(placement: .topBarTrailing) {
                    Button("Save") {
                        store.addToPortfolio(card, condition: condition, pricePaid: pricePaid, quantity: quantity, purchaseDate: purchaseDate)
                        dismiss()
                    }
                    .fontWeight(.semibold)
                    .tint(Theme.accent)
                }
            }
        }
        .preferredColorScheme(.dark)
    }

    private var header: some View {
        HStack(spacing: 14) {
            CardThumbnail(card: card, width: 70)
            VStack(alignment: .leading, spacing: 5) {
                Text(card.name)
                    .font(.system(size: 18, weight: .bold))
                    .foregroundStyle(Theme.textPrimary)
                Text("\(card.setName) · \(card.number)")
                    .font(.system(size: 13))
                    .foregroundStyle(Theme.textSecondary)
                Text("Market \(card.marketPrice.asCurrency)")
                    .font(.system(size: 14, weight: .semibold, design: .rounded))
                    .foregroundStyle(Theme.accent)
            }
            Spacer()
        }
        .padding(16)
        .folioSurface()
    }

    private var conditionSection: some View {
        VStack(alignment: .leading, spacing: 12) {
            sectionLabel("CONDITION")
            LazyVGrid(columns: [GridItem(.adaptive(minimum: 100), spacing: 8)], spacing: 8) {
                ForEach(CardCondition.allCases) { c in
                    Button {
                        withAnimation(.snappy) { condition = c }
                    } label: {
                        Text(c.rawValue)
                            .font(.system(size: 13, weight: .semibold))
                            .frame(maxWidth: .infinity)
                            .padding(.vertical, 11)
                            .background(condition == c ? Theme.accent : Theme.surfaceElevated)
                            .foregroundStyle(condition == c ? Theme.background : Theme.textSecondary)
                            .clipShape(.rect(cornerRadius: 11))
                    }
                    .buttonStyle(.plain)
                }
            }
        }
        .padding(16)
        .folioSurface()
    }

    private var purchaseSection: some View {
        VStack(alignment: .leading, spacing: 16) {
            sectionLabel("PURCHASE DETAILS")
            HStack {
                Text("What I paid")
                    .font(.system(size: 15)).foregroundStyle(Theme.textPrimary)
                Spacer()
                HStack(spacing: 2) {
                    Text("$").foregroundStyle(Theme.textSecondary)
                    TextField("0.00", text: $pricePaidText)
                        .keyboardType(.decimalPad)
                        .multilineTextAlignment(.trailing)
                        .font(.system(size: 16, weight: .semibold, design: .rounded))
                        .foregroundStyle(Theme.textPrimary)
                        .frame(width: 90)
                }
            }
            Divider().overlay(Theme.stroke)
            Stepper(value: $quantity, in: 1...99) {
                HStack {
                    Text("Quantity").font(.system(size: 15)).foregroundStyle(Theme.textPrimary)
                    Spacer()
                    Text("\(quantity)").font(.system(size: 16, weight: .semibold, design: .rounded)).foregroundStyle(Theme.accent)
                }
            }
            Divider().overlay(Theme.stroke)
            DatePicker("Purchase date", selection: $purchaseDate, displayedComponents: .date)
                .font(.system(size: 15))
                .tint(Theme.accent)
        }
        .padding(16)
        .folioSurface()
    }

    private var valuePreview: some View {
        HStack {
            VStack(alignment: .leading, spacing: 4) {
                Text("Current value (\(condition.shortLabel))")
                    .font(.system(size: 12)).foregroundStyle(Theme.textSecondary)
                Text((adjustedValue * Double(quantity)).asCurrency)
                    .font(.system(size: 22, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
            }
            Spacer()
            if pricePaid > 0 {
                let pnl = (adjustedValue - pricePaid) * Double(quantity)
                PnLBadge(amount: pnl, percent: pricePaid > 0 ? (adjustedValue - pricePaid) / pricePaid * 100 : 0)
            }
        }
        .padding(16)
        .folioSurface(Theme.cornerMedium)
        .overlay(
            RoundedRectangle(cornerRadius: Theme.cornerMedium)
                .strokeBorder(Theme.accent.opacity(0.4), lineWidth: 1)
        )
    }

    private func sectionLabel(_ text: String) -> some View {
        Text(text)
            .font(.system(size: 11, weight: .bold))
            .foregroundStyle(Theme.textTertiary)
            .tracking(0.8)
    }
}
