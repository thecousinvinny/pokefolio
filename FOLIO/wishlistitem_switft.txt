import Foundation

/// A card the user is monitoring to buy.
nonisolated struct WishlistItem: Codable, Identifiable, Hashable {
    var id: UUID = UUID()
    var card: Card
    var targetPrice: Double
    var priority: WishlistPriority
    var dateAdded: Date

    /// True when the live market price has reached or dropped below the target.
    var alertTriggered: Bool { card.marketPrice <= targetPrice }

    /// How far the current price is from the target, as a percentage.
    var distanceToTarget: Double {
        guard targetPrice != 0 else { return 0 }
        return (card.marketPrice - targetPrice) / targetPrice * 100
    }
}
