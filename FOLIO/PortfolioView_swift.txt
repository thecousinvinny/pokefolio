import SwiftUI

struct PortfolioView: View {
    @Environment(PortfolioStore.self) private var store
    @State private var sort: PortfolioSort = .value
    @State private var showSearch = false
    @State private var selectedCard: Card?

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.background.ignoresSafeArea()
                if store.portfolio.isEmpty {
                    emptyState
                } else {
                    list
                }
            }
            .navigationTitle("Portfolio")
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    Button { showSearch = true } label: {
                        Image(systemName: "plus")
                    }.tint(Theme.accent)
                }
            }
            .navigationDestination(for: PortfolioItem.self) { item in
                CardDetailView(item: item)
            }
        }
        .preferredColorScheme(.dark)
        .sheet(isPresented: $showSearch) {
            CardSearchView { selectedCard = $0 }
        }
        .sheet(item: $selectedCard) { card in
            AddCardForm(card: card)
        }
    }

    private var list: some View {
        ScrollView {
            VStack(spacing: 14) {
                summaryHeader
                sortPicker
                LazyVStack(spacing: 10) {
                    ForEach(store.sortedPortfolio(sort)) { item in
                        NavigationLink(value: item) {
                            PortfolioRow(item: item)
                        }
                        .buttonStyle(.plain)
                    }
                }
            }
            .padding(.horizontal, 16)
            .padding(.top, 4)
            .padding(.bottom, 32)
        }
        .refreshable { await store.refreshPrices() }
    }

    private var summaryHeader: some View {
        HStack {
            VStack(alignment: .leading, spacing: 4) {
                Text("Binder value")
                    .font(.system(size: 12)).foregroundStyle(Theme.textSecondary)
                Text(store.totalValue.asCurrency)
                    .font(.system(size: 28, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
            }
            Spacer()
            PnLBadge(amount: store.unrealizedPnL, percent: store.unrealizedPercent)
        }
        .padding(16)
        .folioSurface()
    }

    private var sortPicker: some View {
        ScrollView(.horizontal, showsIndicators: false) {
            HStack(spacing: 8) {
                ForEach(PortfolioSort.allCases) { option in
                    Button {
                        withAnimation(.snappy) { sort = option }
                    } label: {
                        Text(option.rawValue)
                            .font(.system(size: 13, weight: .semibold))
                            .padding(.horizontal, 14).padding(.vertical, 8)
                            .background(sort == option ? Theme.accent : Theme.surface)
                            .foregroundStyle(sort == option ? Theme.background : Theme.textSecondary)
                            .clipShape(Capsule())
                    }
                    .buttonStyle(.plain)
                }
            }
        }
        .contentMargins(.horizontal, 0)
    }

    private var emptyState: some View {
        VStack(spacing: 16) {
            Image(systemName: "square.stack.3d.up.slash")
                .font(.system(size: 48, weight: .light))
                .foregroundStyle(Theme.textTertiary)
            Text("Your binder is empty")
                .font(.system(size: 18, weight: .semibold))
                .foregroundStyle(Theme.textPrimary)
            Text("Scan or search a card to start\ntracking its live value.")
                .multilineTextAlignment(.center)
                .font(.system(size: 14))
                .foregroundStyle(Theme.textSecondary)
            Button {
                showSearch = true
            } label: {
                Label("Add a card", systemImage: "plus")
                    .font(.system(size: 15, weight: .semibold))
                    .foregroundStyle(Theme.background)
                    .padding(.horizontal, 22).padding(.vertical, 12)
                    .background(Theme.accent)
                    .clipShape(Capsule())
            }
            .padding(.top, 4)
        }
        .padding(40)
    }
}

private struct PortfolioRow: View {
    let item: PortfolioItem

    var body: some View {
        HStack(spacing: 12) {
            CardThumbnail(card: item.card, width: 50)
            VStack(alignment: .leading, spacing: 4) {
                Text(item.card.name)
                    .font(.system(size: 15, weight: .semibold))
                    .foregroundStyle(Theme.textPrimary)
                    .lineLimit(1)
                HStack(spacing: 6) {
                    Text(item.condition.shortLabel)
                        .font(.system(size: 10, weight: .bold))
                        .padding(.horizontal, 6).padding(.vertical, 2)
                        .background(Theme.surfaceElevated)
                        .foregroundStyle(Theme.textSecondary)
                        .clipShape(.rect(cornerRadius: 5))
                    if item.quantity > 1 {
                        Text("×\(item.quantity)")
                            .font(.system(size: 11, weight: .semibold))
                            .foregroundStyle(Theme.textTertiary)
                    }
                    Text(item.card.setName)
                        .font(.system(size: 11))
                        .foregroundStyle(Theme.textTertiary)
                        .lineLimit(1)
                }
            }
            Spacer()
            VStack(alignment: .trailing, spacing: 5) {
                Text(item.currentValue.asCurrency)
                    .font(.system(size: 15, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
                PnLBadge(amount: 0, percent: item.card.dailyChangePercent, compact: true)
            }
        }
        .padding(12)
        .folioSurface(Theme.cornerMedium)
    }
}
