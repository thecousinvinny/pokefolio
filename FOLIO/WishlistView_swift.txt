import SwiftUI

struct WishlistView: View {
    @Environment(PortfolioStore.self) private var store
    @State private var showSearch = false
    @State private var selectedCard: Card?

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.background.ignoresSafeArea()
                if store.wishlist.isEmpty {
                    emptyState
                } else {
                    list
                }
            }
            .navigationTitle("Wishlist")
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    Button { showSearch = true } label: {
                        Image(systemName: "plus")
                    }.tint(Theme.accent)
                }
            }
        }
        .preferredColorScheme(.dark)
        .sheet(isPresented: $showSearch) {
            CardSearchView { selectedCard = $0 }
        }
        .sheet(item: $selectedCard) { card in
            AddWishlistForm(card: card)
        }
    }

    private var list: some View {
        ScrollView {
            LazyVStack(spacing: 10) {
                ForEach(store.wishlist) { item in
                    WishlistRow(item: item)
                        .swipeActions {
                            Button(role: .destructive) {
                                store.removeFromWishlist(item)
                            } label: { Label("Remove", systemImage: "trash") }
                        }
                        .contextMenu {
                            Button(role: .destructive) {
                                store.removeFromWishlist(item)
                            } label: { Label("Remove", systemImage: "trash") }
                        }
                }
            }
            .padding(.horizontal, 16)
            .padding(.top, 4)
            .padding(.bottom, 32)
        }
        .refreshable { await store.refreshPrices() }
    }

    private var emptyState: some View {
        VStack(spacing: 16) {
            Image(systemName: "star.slash")
                .font(.system(size: 48, weight: .light))
                .foregroundStyle(Theme.textTertiary)
            Text("No chase cards yet")
                .font(.system(size: 18, weight: .semibold))
                .foregroundStyle(Theme.textPrimary)
            Text("Add cards you want and set a target\nprice to get notified on dips.")
                .multilineTextAlignment(.center)
                .font(.system(size: 14))
                .foregroundStyle(Theme.textSecondary)
            Button { showSearch = true } label: {
                Label("Add a card", systemImage: "plus")
                    .font(.system(size: 15, weight: .semibold))
                    .foregroundStyle(Theme.background)
                    .padding(.horizontal, 22).padding(.vertical, 12)
                    .background(Theme.accent)
                    .clipShape(Capsule())
            }
            .padding(.top, 4)
        }
        .padding(40)
    }
}

private struct WishlistRow: View {
    let item: WishlistItem

    var body: some View {
        VStack(spacing: 12) {
            HStack(spacing: 12) {
                CardThumbnail(card: item.card, width: 50)
                VStack(alignment: .leading, spacing: 4) {
                    HStack(spacing: 6) {
                        Circle().fill(item.priority.color).frame(width: 7, height: 7)
                        Text(item.card.name)
                            .font(.system(size: 15, weight: .semibold))
                            .foregroundStyle(Theme.textPrimary)
                            .lineLimit(1)
                    }
                    Text(item.card.setName)
                        .font(.system(size: 12))
                        .foregroundStyle(Theme.textTertiary)
                        .lineLimit(1)
                }
                Spacer()
                VStack(alignment: .trailing, spacing: 3) {
                    Text(item.card.marketPrice.asCurrency)
                        .font(.system(size: 15, weight: .bold, design: .rounded))
                        .foregroundStyle(Theme.textPrimary)
                    Text("Target \(item.targetPrice.asCurrency)")
                        .font(.system(size: 11, weight: .medium))
                        .foregroundStyle(Theme.textSecondary)
                }
            }

            HStack(spacing: 10) {
                Sparkline(values: item.card.priceHistory,
                          color: item.card.dailyChange >= 0 ? Theme.gain : Theme.loss,
                          showFill: false, lineWidth: 1.5)
                    .frame(height: 28)
                if item.alertTriggered {
                    Label("Target hit", systemImage: "checkmark.circle.fill")
                        .font(.system(size: 11, weight: .bold))
                        .foregroundStyle(Theme.gain)
                        .padding(.horizontal, 8).padding(.vertical, 4)
                        .background(Theme.gain.opacity(0.14))
                        .clipShape(Capsule())
                } else {
                    Text(String(format: "%.0f%% to go", item.distanceToTarget))
                        .font(.system(size: 11, weight: .semibold))
                        .foregroundStyle(Theme.textTertiary)
                        .padding(.horizontal, 8).padding(.vertical, 4)
                        .background(Theme.surfaceElevated)
                        .clipShape(Capsule())
                }
            }
        }
        .padding(12)
        .folioSurface(Theme.cornerMedium)
        .overlay(
            RoundedRectangle(cornerRadius: Theme.cornerMedium)
                .strokeBorder(item.alertTriggered ? Theme.gain.opacity(0.4) : Color.clear, lineWidth: 1)
        )
    }
}
