//  CardSearchView.swift
//  Folio
//
//  Live card search backed by the Pokémon TCG API. Used as the first step
//  of both "Add to Portfolio" and "Add to Wishlist". A camera-scan entry
//  point is provided for real-device use.
//

import SwiftUI

struct CardSearchView: View {
    let onSelect: (Card) -> Void

    @Environment(\.dismiss) private var dismiss
    @State private var query = ""
    @State private var results: [Card] = []
    @State private var isLoading = false
    @State private var searchTask: Task<Void, Never>?
    @State private var showScanner = false

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.background.ignoresSafeArea()
                content
            }
            .navigationTitle("Find a Card")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .topBarLeading) {
                    Button("Cancel") { dismiss() }.tint(Theme.textSecondary)
                }
                ToolbarItem(placement: .topBarTrailing) {
                    Button { showScanner = true } label: {
                        Image(systemName: "camera.viewfinder")
                    }.tint(Theme.accent)
                }
            }
            .searchable(text: $query, placement: .navigationBarDrawer(displayMode: .always), prompt: "Search Charizard, Umbreon…")
            .onChange(of: query) { _, newValue in scheduleSearch(newValue) }
            .sheet(isPresented: $showScanner) { ScanPlaceholderView() }
        }
        .preferredColorScheme(.dark)
    }

    @ViewBuilder
    private var content: some View {
        if isLoading {
            ProgressView().controlSize(.large).tint(Theme.accent)
        } else if results.isEmpty {
            emptyState
        } else {
            ScrollView {
                LazyVStack(spacing: 10) {
                    ForEach(results) { card in
                        Button {
                            onSelect(card)
                            dismiss()
                        } label: {
                            CardSearchRow(card: card)
                        }
                        .buttonStyle(.plain)
                    }
                }
                .padding(.horizontal, 16)
                .padding(.top, 8)
            }
        }
    }

    private var emptyState: some View {
        VStack(spacing: 14) {
            Image(systemName: query.isEmpty ? "magnifyingglass" : "questionmark.folder")
                .font(.system(size: 40, weight: .light))
                .foregroundStyle(Theme.textTertiary)
            Text(query.isEmpty ? "Search by card name to pull live\nTCGPlayer pricing and artwork." : "No cards found for “\(query)”.")
                .multilineTextAlignment(.center)
                .font(.system(size: 14))
                .foregroundStyle(Theme.textSecondary)
        }
        .padding(40)
    }

    private func scheduleSearch(_ text: String) {
        searchTask?.cancel()
        let trimmed = text.trimmingCharacters(in: .whitespaces)
        guard trimmed.count >= 2 else {
            results = []
            isLoading = false
            return
        }
        searchTask = Task {
            try? await Task.sleep(for: .milliseconds(350))
            guard !Task.isCancelled else { return }
            isLoading = true
            let found = (try? await PokemonTCGService.shared.search(trimmed)) ?? []
            guard !Task.isCancelled else { return }
            results = found
            isLoading = false
        }
    }
}

private struct CardSearchRow: View {
    let card: Card

    var body: some View {
        HStack(spacing: 12) {
            CardThumbnail(card: card, width: 48)
            VStack(alignment: .leading, spacing: 3) {
                Text(card.name)
                    .font(.system(size: 15, weight: .semibold))
                    .foregroundStyle(Theme.textPrimary)
                    .lineLimit(1)
                Text("\(card.setName) · \(card.number)")
                    .font(.system(size: 12))
                    .foregroundStyle(Theme.textSecondary)
                    .lineLimit(1)
            }
            Spacer()
            Text(card.marketPrice.asCurrency)
                .font(.system(size: 15, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.accent)
        }
        .padding(12)
        .folioSurface(Theme.cornerMedium)
    }
}

/// Camera scanning is unavailable in the cloud simulator.
private struct ScanPlaceholderView: View {
    @Environment(\.dismiss) private var dismiss

    var body: some View {
        ZStack {
            Theme.background.ignoresSafeArea()
            VStack(spacing: 18) {
                Image(systemName: "camera.viewfinder")
                    .font(.system(size: 54, weight: .light))
                    .foregroundStyle(Theme.accent)
                Text("Scan to Add")
                    .font(.system(size: 20, weight: .bold))
                    .foregroundStyle(Theme.textPrimary)
                Text("Install this app on your device via the Rork App to scan cards with your camera.")
                    .multilineTextAlignment(.center)
                    .font(.system(size: 14))
                    .foregroundStyle(Theme.textSecondary)
                    .padding(.horizontal, 32)
                Button("Got it") { dismiss() }
                    .font(.system(size: 15, weight: .semibold))
                    .foregroundStyle(Theme.background)
                    .padding(.horizontal, 28).padding(.vertical, 12)
                    .background(Theme.accent)
                    .clipShape(Capsule())
            }
        }
        .preferredColorScheme(.dark)
    }
}
