import SwiftUI

struct CardDetailView: View {
    let item: PortfolioItem
    @Environment(PortfolioStore.self) private var store
    @Environment(\.dismiss) private var dismiss
    @State private var showSell = false

    /// Pulls the freshest copy from the store so prices stay live.
    private var live: PortfolioItem {
        store.portfolio.first { $0.id == item.id } ?? item
    }

    private var isFavorite: Bool { store.favoriteCardID == item.card.id }

    var body: some View {
        ZStack {
            Theme.background.ignoresSafeArea()
            ScrollView {
                VStack(spacing: 20) {
                    artwork
                    valueBlock
                    priceTiers
                    holdingDetails
                    actions
                }
                .padding(16)
                .padding(.bottom, 24)
            }
        }
        .navigationTitle(item.card.name)
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .topBarTrailing) {
                Button {
                    store.setFavorite(item)
                } label: {
                    Image(systemName: isFavorite ? "heart.fill" : "heart")
                        .foregroundStyle(isFavorite ? Theme.violet : Theme.textSecondary)
                }
            }
        }
        .sheet(isPresented: $showSell) {
            LogSaleView(item: live)
        }
        .preferredColorScheme(.dark)
    }

    private var artwork: some View {
        CardThumbnail(card: item.card, width: 220, cornerRadius: 16)
            .shadow(color: .black.opacity(0.5), radius: 18, y: 10)
            .padding(.top, 8)
    }

    private var valueBlock: some View {
        VStack(spacing: 8) {
            Text(live.currentValue.asCurrency)
                .font(.system(size: 38, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            PnLBadge(amount: live.unrealizedPnL, percent: live.pnlPercent)
            Sparkline(values: item.card.priceHistory, color: live.card.dailyChange >= 0 ? Theme.gain : Theme.loss)
                .frame(height: 60)
                .padding(.top, 6)
        }
        .frame(maxWidth: .infinity)
        .padding(20)
        .folioSurface()
    }

    private var priceTiers: some View {
        VStack(spacing: 0) {
            tierRow("Market Price", live.card.marketPrice, Theme.accent, "Best for valuation")
            divider
            tierRow("Listed Median", live.card.listedMedian, Theme.violet, "Middle of active listings")
            divider
            tierRow("Direct Low", live.card.directLow, Theme.gold, "Cheapest Near Mint copy")
        }
        .padding(16)
        .folioSurface()
    }

    private func tierRow(_ label: String, _ value: Double, _ tint: Color, _ note: String) -> some View {
        HStack {
            Circle().fill(tint).frame(width: 8, height: 8)
            VStack(alignment: .leading, spacing: 1) {
                Text(label).font(.system(size: 14, weight: .semibold)).foregroundStyle(Theme.textPrimary)
                Text(note).font(.system(size: 11)).foregroundStyle(Theme.textTertiary)
            }
            Spacer()
            Text(value.asCurrency)
                .font(.system(size: 16, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
        }
        .padding(.vertical, 10)
    }

    private var holdingDetails: some View {
        VStack(spacing: 0) {
            detailRow("Condition", item.condition.rawValue)
            divider
            detailRow("Quantity", "\(item.quantity)")
            divider
            detailRow("Paid", item.invested.asCurrency)
            divider
            detailRow("Purchased", item.purchaseDate.formatted(date: .abbreviated, time: .omitted))
            divider
            detailRow("Set", "\(item.card.setName) · \(item.card.number)")
        }
        .padding(16)
        .folioSurface()
    }

    private func detailRow(_ label: String, _ value: String) -> some View {
        HStack {
            Text(label).font(.system(size: 14)).foregroundStyle(Theme.textSecondary)
            Spacer()
            Text(value).font(.system(size: 14, weight: .semibold)).foregroundStyle(Theme.textPrimary)
        }
        .padding(.vertical, 11)
    }

    private var actions: some View {
        VStack(spacing: 10) {
            Button {
                showSell = true
            } label: {
                Label("Log Sale / Trade", systemImage: "arrow.up.right.circle.fill")
                    .font(.system(size: 16, weight: .semibold))
                    .foregroundStyle(Theme.background)
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 14)
                    .background(Theme.accent)
                    .clipShape(.rect(cornerRadius: 14))
            }
            Button(role: .destructive) {
                store.removeFromPortfolio(item)
                dismiss()
            } label: {
                Text("Remove from Portfolio")
                    .font(.system(size: 15, weight: .medium))
                    .foregroundStyle(Theme.loss)
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 13)
                    .folioSurface(14)
            }
        }
    }

    private var divider: some View { Divider().overlay(Theme.stroke) }
}
