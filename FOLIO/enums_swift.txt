import SwiftUI

/// Physical condition of a card, with a multiplier applied to market price.
nonisolated enum CardCondition: String, Codable, CaseIterable, Identifiable {
    case mint = "Mint"
    case nearMint = "Near Mint"
    case lightlyPlayed = "Lightly Played"
    case moderatelyPlayed = "Moderately Played"
    case heavilyPlayed = "Heavily Played"
    case damaged = "Damaged"

    var id: String { rawValue }

    /// Rough market multiplier relative to a Near Mint copy.
    var priceMultiplier: Double {
        switch self {
        case .mint: return 1.08
        case .nearMint: return 1.0
        case .lightlyPlayed: return 0.85
        case .moderatelyPlayed: return 0.7
        case .heavilyPlayed: return 0.55
        case .damaged: return 0.4
        }
    }

    var shortLabel: String {
        switch self {
        case .mint: return "M"
        case .nearMint: return "NM"
        case .lightlyPlayed: return "LP"
        case .moderatelyPlayed: return "MP"
        case .heavilyPlayed: return "HP"
        case .damaged: return "DMG"
        }
    }
}

/// Wishlist priority for strategic buying.
nonisolated enum WishlistPriority: String, Codable, CaseIterable, Identifiable {
    case high = "High Priority"
    case medium = "Medium"
    case low = "Low Priority"

    var id: String { rawValue }

    var color: Color {
        switch self {
        case .high: return Theme.loss
        case .medium: return Theme.gold
        case .low: return Theme.textSecondary
        }
    }
}

/// Sorting options for the portfolio binder.
enum PortfolioSort: String, CaseIterable, Identifiable {
    case gainers = "Gainers Today"
    case value = "Highest Value"
    case recent = "Recently Added"

    var id: String { rawValue }
}
