import SwiftUI

/// Condition + price-paid popup when adding a browse card to the portfolio.
struct AddToPortfolioSheet: View {
    @Environment(CollectionStore.self) private var store
    @Environment(\.dismiss) private var dismiss

    let template: Card
    @State private var condition: CardCondition = .nearMint
    @State private var pricePaid: Double

    init(template: Card) {
        self.template = template
        _pricePaid = State(initialValue: (template.marketPrice * 100).rounded() / 100)
    }

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.backgroundGradient.ignoresSafeArea()
                ScrollView {
                    VStack(spacing: 20) {
                        Color.black
                            .frame(width: 130, height: 180)
                            .overlay { CardArtwork(card: template).allowsHitTesting(false) }
                            .clipShape(.rect(cornerRadius: 14))
                            .shadow(color: .black.opacity(0.4), radius: 12, y: 6)

                        Text(template.name)
                            .font(.system(size: 20, weight: .heavy, design: .rounded))
                            .foregroundStyle(Theme.textPrimary)

                        VStack(alignment: .leading, spacing: 12) {
                            Text("CONDITION")
                                .font(.system(size: 11, weight: .bold, design: .rounded))
                                .tracking(1)
                                .foregroundStyle(Theme.textTertiary)
                            Picker("Condition", selection: $condition) {
                                ForEach(CardCondition.allCases) { c in
                                    Text(c.rawValue).tag(c)
                                }
                            }
                            .pickerStyle(.segmented)

                            Text(condition.fullName + String(format: " · %.0f%% of market", condition.priceMultiplier * 100))
                                .font(.system(size: 12))
                                .foregroundStyle(Theme.textSecondary)
                        }
                        .padding(16)
                        .background(Theme.surface)
                        .clipShape(.rect(cornerRadius: 14))

                        HStack {
                            Text("Price Paid")
                                .font(.system(size: 15, weight: .semibold, design: .rounded))
                                .foregroundStyle(Theme.textPrimary)
                            Spacer()
                            HStack(spacing: 2) {
                                Text("$").foregroundStyle(Theme.textSecondary)
                                TextField("0", value: $pricePaid, format: .number.precision(.fractionLength(2)))
                                    .keyboardType(.decimalPad)
                                    .multilineTextAlignment(.trailing)
                                    .frame(width: 100)
                                    .foregroundStyle(Theme.gold)
                                    .font(.system(size: 18, weight: .bold, design: .rounded))
                            }
                        }
                        .padding(16)
                        .background(Theme.surface)
                        .clipShape(.rect(cornerRadius: 14))

                        Button {
                            store.addToPortfolio(template, condition: condition, pricePaid: pricePaid)
                            UINotificationFeedbackGenerator().notificationOccurred(.success)
                            dismiss()
                        } label: {
                            Text("Add to Portfolio")
                                .font(.system(size: 17, weight: .bold, design: .rounded))
                                .frame(maxWidth: .infinity)
                                .padding(.vertical, 16)
                                .background(LinearGradient(colors: [Theme.emerald, Theme.sky], startPoint: .leading, endPoint: .trailing))
                                .foregroundStyle(.white)
                                .clipShape(.rect(cornerRadius: 14))
                        }
                    }
                    .padding(20)
                }
            }
            .navigationTitle("Add Card")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }.foregroundStyle(Theme.textSecondary)
                }
            }
            .toolbarBackground(Theme.bg, for: .navigationBar)
        }
    }
}
