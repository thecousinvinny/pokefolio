//
//  CollectionStore.swift
//  PokeVault
//
//  Single source of truth for the user's collection, wishlist, and ledger.
//  Persists to UserDefaults as JSON.
//

import Foundation
import SwiftUI

@Observable
final class CollectionStore {
    /// All cards the user has saved (owned, forSale, wishlist).
    var cards: [Card] = []
    /// Completed sales/trades.
    var soldEntries: [SoldEntry] = []
    /// The favorite card id, surfaced large on the dashboard.
    var favoriteCardID: UUID?

    private let cardsKey = "pv.cards.v1"
    private let soldKey = "pv.sold.v1"
    private let favKey = "pv.favorite.v1"
    private let seededKey = "pv.seeded.v1"

    init() {
        load()
    }

    // MARK: - Derived collections

    var ownedCards: [Card] { cards.filter { $0.status == .owned || $0.status == .forSale } }
    var wishlistCards: [Card] { cards.filter { $0.status == .wishlist } }

    var totalPortfolioValue: Double {
        ownedCards.reduce(0) { $0 + $1.conditionAdjustedValue }
    }

    var totalCostBasis: Double {
        ownedCards.reduce(0) { $0 + ($1.pricePaid ?? 0) }
    }

    var unrealizedProfit: Double { totalPortfolioValue - totalCostBasis }

    var lifetimeEarnings: Double {
        soldEntries.reduce(0) { $0 + $1.netProfit }
    }

    var bestSale: SoldEntry? {
        soldEntries.max { $0.netProfit < $1.netProfit }
    }

    /// Top owned cards by 7-day gain.
    var marketMovers: [Card] {
        ownedCards
            .filter { $0.weeklyChangePercent > 0 }
            .sorted { $0.weeklyChangePercent > $1.weeklyChangePercent }
            .prefix(3)
            .map { $0 }
    }

    var favoriteCard: Card? {
        guard let id = favoriteCardID else { return nil }
        return cards.first { $0.id == id }
    }

    /// Profit grouped by month for the last 6 months (oldest first).
    var profitByMonth: [(label: String, profit: Double)] {
        var buckets: [(label: String, profit: Double)] = []
        let cal = Calendar.current
        let fmt = DateFormatter()
        fmt.dateFormat = "MMM"
        for offset in stride(from: 5, through: 0, by: -1) {
            guard let date = cal.date(byAdding: .month, value: -offset, to: Date()) else { continue }
            let comps = cal.dateComponents([.year, .month], from: date)
            let total = soldEntries
                .filter {
                    let c = cal.dateComponents([.year, .month], from: $0.dateSold)
                    return c.year == comps.year && c.month == comps.month
                }
                .reduce(0) { $0 + $1.netProfit }
            buckets.append((label: fmt.string(from: date), profit: total))
        }
        return buckets
    }

    // MARK: - Mutations

    func contains(catalogName: String, status: CardStatus) -> Bool {
        cards.contains { $0.name == catalogName && $0.status == status }
    }

    func addToPortfolio(_ template: Card, condition: CardCondition, pricePaid: Double) {
        var c = template
        c.id = UUID()
        c.status = .owned
        c.condition = condition
        c.pricePaid = pricePaid
        c.dateAdded = Date()
        cards.append(c)
        save()
    }

    func addToWishlist(_ template: Card, target: Double?) {
        var c = template
        c.id = UUID()
        c.status = .wishlist
        c.targetPrice = target
        c.alertsEnabled = target != nil
        c.dateAdded = Date()
        cards.append(c)
        save()
    }

    func update(_ card: Card) {
        guard let idx = cards.firstIndex(where: { $0.id == card.id }) else { return }
        cards[idx] = card
        save()
    }

    func remove(_ card: Card) {
        cards.removeAll { $0.id == card.id }
        if favoriteCardID == card.id { favoriteCardID = nil }
        save()
    }

    func setFavorite(_ card: Card) {
        favoriteCardID = favoriteCardID == card.id ? nil : card.id
        save()
    }

    func markForSale(_ ids: Set<UUID>) {
        for id in ids {
            guard let idx = cards.firstIndex(where: { $0.id == id }) else { continue }
            cards[idx].status = cards[idx].status == .forSale ? .owned : .forSale
        }
        save()
    }

    /// Completes a sale, moving the card out of the collection into the ledger.
    func sell(_ card: Card, soldPrice: Double, fees: Double, shipping: Double) {
        var sold = card
        sold.status = .sold
        let entry = SoldEntry(
            card: sold,
            dateSold: Date(),
            soldPrice: soldPrice,
            fees: fees,
            shipping: shipping,
            costBasis: card.pricePaid ?? 0
        )
        soldEntries.insert(entry, at: 0)
        remove(card)
        save()
    }

    /// Re-adds a previously sold card back to the portfolio.
    func reprint(_ entry: SoldEntry) {
        var c = entry.card
        c.id = UUID()
        c.status = .owned
        c.pricePaid = entry.soldPrice
        c.dateAdded = Date()
        cards.append(c)
        save()
    }

    /// Moves every wishlist card into the portfolio (bulk lot purchase).
    func moveAllWishlistToPortfolio() {
        for idx in cards.indices where cards[idx].status == .wishlist {
            cards[idx].status = .owned
            cards[idx].pricePaid = cards[idx].targetPrice ?? cards[idx].marketPrice
            cards[idx].condition = .nearMint
            cards[idx].dateAdded = Date()
        }
        save()
    }

    /// Clears alerts on wishlist cards whose price exceeds their target.
    func clearSpikedAlerts() {
        for idx in cards.indices where cards[idx].status == .wishlist {
            if let target = cards[idx].targetPrice, cards[idx].marketPrice > target {
                cards[idx].alertsEnabled = false
            }
        }
        save()
    }

    // MARK: - Persistence

    private func save() {
        let enc = JSONEncoder()
        if let d = try? enc.encode(cards) { UserDefaults.standard.set(d, forKey: cardsKey) }
        if let d = try? enc.encode(soldEntries) { UserDefaults.standard.set(d, forKey: soldKey) }
        if let id = favoriteCardID { UserDefaults.standard.set(id.uuidString, forKey: favKey) }
        else { UserDefaults.standard.removeObject(forKey: favKey) }
    }

    private func load() {
        let dec = JSONDecoder()
        let defaults = UserDefaults.standard

        if defaults.bool(forKey: seededKey) {
            if let d = defaults.data(forKey: cardsKey), let c = try? dec.decode([Card].self, from: d) {
                cards = c
            }
            if let d = defaults.data(forKey: soldKey), let s = try? dec.decode([SoldEntry].self, from: d) {
                soldEntries = s
            }
            if let s = defaults.string(forKey: favKey) { favoriteCardID = UUID(uuidString: s) }
        } else {
            let (seedCards, seedSold) = SampleData.seedCollection()
            cards = seedCards
            soldEntries = seedSold
            favoriteCardID = seedCards.first(where: { $0.name == "Charizard ex" })?.id
            defaults.set(true, forKey: seededKey)
            save()
        }
    }
}
