import SwiftUI

struct SoldView: View {
    @Environment(CollectionStore.self) private var store

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.backgroundGradient.ignoresSafeArea()
                if store.soldEntries.isEmpty {
                    emptyState
                } else {
                    ScrollView {
                        VStack(spacing: 18) {
                            summaryCard
                            monthlyChart
                            ledger
                        }
                        .padding(.horizontal, 16)
                        .padding(.top, 8)
                        .padding(.bottom, 40)
                    }
                }
            }
            .navigationTitle("Sold / Traded")
            .toolbarBackground(Theme.bg, for: .navigationBar)
        }
    }

    private var summaryCard: some View {
        VStack(spacing: 16) {
            VStack(spacing: 4) {
                Text("LIFETIME NET PROFIT")
                    .font(.system(size: 11, weight: .bold, design: .rounded)).tracking(1.5)
                    .foregroundStyle(Theme.textSecondary)
                Text(formatPrice(store.lifetimeEarnings))
                    .font(.system(size: 46, weight: .heavy, design: .rounded))
                    .foregroundStyle(Theme.profitColor(store.lifetimeEarnings))
            }
            if let best = store.bestSale {
                HStack(spacing: 10) {
                    Image(systemName: "trophy.fill").foregroundStyle(Theme.gold)
                    VStack(alignment: .leading, spacing: 1) {
                        Text("Best sale ever")
                            .font(.system(size: 11, weight: .semibold)).foregroundStyle(Theme.textTertiary)
                        Text(best.card.name)
                            .font(.system(size: 14, weight: .bold, design: .rounded))
                            .foregroundStyle(Theme.textPrimary)
                    }
                    Spacer()
                    Text(formatPrice(best.netProfit))
                        .font(.system(size: 18, weight: .heavy, design: .rounded))
                        .foregroundStyle(Theme.emerald)
                }
                .padding(12)
                .background(Theme.bgElevated)
                .clipShape(.rect(cornerRadius: 14))
            }
        }
        .padding(20)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 22))
    }

    private var monthlyChart: some View {
        let data = store.profitByMonth
        let maxVal = max(data.map { abs($0.profit) }.max() ?? 1, 1)
        return VStack(alignment: .leading, spacing: 14) {
            Text("Profit per Month")
                .font(.system(size: 15, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            HStack(alignment: .bottom, spacing: 10) {
                ForEach(Array(data.enumerated()), id: \.offset) { _, item in
                    VStack(spacing: 6) {
                        Text(item.profit != 0 ? formatPrice(item.profit, compact: true) : "")
                            .font(.system(size: 9, weight: .bold, design: .rounded))
                            .foregroundStyle(Theme.profitColor(item.profit))
                            .lineLimit(1).minimumScaleFactor(0.7)
                        RoundedRectangle(cornerRadius: 6)
                            .fill(
                                LinearGradient(
                                    colors: item.profit >= 0
                                        ? [Theme.emerald, Theme.emerald.opacity(0.5)]
                                        : [Theme.crimson, Theme.crimson.opacity(0.5)],
                                    startPoint: .top, endPoint: .bottom
                                )
                            )
                            .frame(height: max(6, CGFloat(abs(item.profit) / maxVal) * 120))
                        Text(item.label)
                            .font(.system(size: 11, weight: .semibold))
                            .foregroundStyle(Theme.textSecondary)
                    }
                    .frame(maxWidth: .infinity)
                }
            }
            .frame(height: 170, alignment: .bottom)
        }
        .padding(18)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 20))
    }

    private var ledger: some View {
        VStack(alignment: .leading, spacing: 10) {
            Text("History")
                .font(.system(size: 15, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
                .padding(.horizontal, 4)
            ForEach(store.soldEntries) { entry in
                SoldRow(entry: entry)
            }
        }
    }

    private var emptyState: some View {
        VStack(spacing: 20) {
            ZStack {
                Circle().fill(Theme.surface).frame(width: 120, height: 120)
                Image(systemName: "chart.bar.xaxis")
                    .font(.system(size: 46)).foregroundStyle(Theme.emerald)
            }
            Text("No sales yet")
                .font(.system(size: 22, weight: .heavy, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            Text("When you sell a card from your Portfolio, your profit ledger appears here.")
                .font(.system(size: 14)).foregroundStyle(Theme.textSecondary)
                .multilineTextAlignment(.center)
        }
        .padding(40)
    }
}

struct SoldRow: View {
    @Environment(CollectionStore.self) private var store
    let entry: SoldEntry

    private var dateText: String {
        let f = DateFormatter()
        f.dateFormat = "MMM d, yyyy"
        return f.string(from: entry.dateSold)
    }

    var body: some View {
        HStack(spacing: 12) {
            Color.black
                .frame(width: 54, height: 75)
                .overlay { CardArtwork(card: entry.card, showHoloSheen: false).allowsHitTesting(false) }
                .clipShape(.rect(cornerRadius: 8))

            VStack(alignment: .leading, spacing: 3) {
                Text(entry.card.name)
                    .font(.system(size: 15, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary).lineLimit(1)
                Text(dateText)
                    .font(.system(size: 11, weight: .medium))
                    .foregroundStyle(Theme.textTertiary)
                HStack(spacing: 8) {
                    Text("Sold \(formatPrice(entry.soldPrice))")
                    Text("· fees \(formatPrice(entry.fees + entry.shipping))")
                }
                .font(.system(size: 11))
                .foregroundStyle(Theme.textSecondary)
            }

            Spacer()

            VStack(alignment: .trailing, spacing: 6) {
                Text(formatPrice(entry.netProfit))
                    .font(.system(size: 17, weight: .heavy, design: .rounded))
                    .foregroundStyle(Theme.profitColor(entry.netProfit))
                Button {
                    store.reprint(entry)
                    UIImpactFeedbackGenerator(style: .light).impactOccurred()
                } label: {
                    Label("Reprint", systemImage: "arrow.uturn.left")
                        .font(.system(size: 10, weight: .bold, design: .rounded))
                        .padding(.horizontal, 8).padding(.vertical, 5)
                        .background(Theme.bgElevated).foregroundStyle(Theme.sky)
                        .clipShape(Capsule())
                }
            }
        }
        .padding(12)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 16))
    }
}
