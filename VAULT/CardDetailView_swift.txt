//  CardDetailView.swift
//  PokeVault
//
//  Near-full-screen "big card" moment with condition slider, paid-vs-market,
//  price graph, and sell/trade action.
//

import SwiftUI

struct CardDetailView: View {
    @Environment(CollectionStore.self) private var store
    @Environment(\.dismiss) private var dismiss

    let cardID: UUID
    @State private var showSellSheet = false
    @State private var tilt: Double = 0

    private var card: Card? { store.cards.first { $0.id == cardID } }

    var body: some View {
        ZStack {
            Theme.backgroundGradient.ignoresSafeArea()

            if let card {
                ScrollView {
                    VStack(spacing: 24) {
                        heroCard(card)
                        priceSummary(card)
                        if card.status == .owned || card.status == .forSale {
                            conditionSection(card)
                        }
                        priceGraph(card)
                        actions(card)
                    }
                    .padding(20)
                    .padding(.bottom, 40)
                }
            } else {
                Text("Card not found")
                    .foregroundStyle(Theme.textSecondary)
            }
        }
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .principal) {
                Text(card?.name ?? "")
                    .font(.system(size: 16, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
            }
        }
        .toolbarBackground(Theme.bg, for: .navigationBar)
        .sheet(isPresented: $showSellSheet) {
            if let card { SellSheet(card: card) }
        }
    }

    private func heroCard(_ card: Card) -> some View {
        Color.black
            .aspectRatio(0.72, contentMode: .fit)
            .frame(maxWidth: 300)
            .overlay {
                CardArtwork(card: card)
                    .allowsHitTesting(false)
            }
            .clipShape(.rect(cornerRadius: 22))
            .overlay(
                RoundedRectangle(cornerRadius: 22)
                    .stroke(Theme.gold.opacity(0.5), lineWidth: 1.5)
            )
            .shadow(color: card.type.artColors.first?.opacity(0.6) ?? .black, radius: 30, y: 16)
            .rotation3DEffect(.degrees(tilt), axis: (x: 0, y: 1, z: 0))
            .onAppear {
                withAnimation(.easeInOut(duration: 2.4).repeatForever(autoreverses: true)) {
                    tilt = 6
                }
            }
            .frame(maxWidth: .infinity)
    }

    private func priceSummary(_ card: Card) -> some View {
        VStack(spacing: 12) {
            HStack {
                metaChip(card.setName, system: "rectangle.stack.fill")
                metaChip(card.rarity.rawValue, system: "star.fill")
                metaChip(card.holo.rawValue, system: "sparkles")
            }
            HStack(spacing: 14) {
                statBox(title: "Market (NM)", value: formatPrice(card.marketPrice), color: Theme.textPrimary)
                if card.status == .owned || card.status == .forSale {
                    statBox(title: "Your Value", value: formatPrice(card.conditionAdjustedValue), color: Theme.gold)
                }
            }
            if let paid = card.pricePaid, card.status != .wishlist {
                HStack(spacing: 14) {
                    statBox(title: "You Paid", value: formatPrice(paid), color: Theme.textSecondary)
                    statBox(
                        title: "Unrealized P/L",
                        value: formatPrice(card.unrealizedProfit),
                        color: Theme.profitColor(card.unrealizedProfit)
                    )
                }
            }
            if card.status == .wishlist, let target = card.targetPrice {
                statBox(title: "Buy if ≤", value: formatPrice(target), color: Theme.violet)
                    .frame(maxWidth: .infinity)
            }
        }
    }

    private func conditionSection(_ card: Card) -> some View {
        VStack(alignment: .leading, spacing: 12) {
            HStack {
                Text("Condition")
                    .font(.system(size: 15, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
                Spacer()
                Text(card.condition.fullName)
                    .font(.system(size: 13, weight: .semibold))
                    .foregroundStyle(Theme.gold)
            }

            let index = Binding<Double>(
                get: { Double(CardCondition.allCases.firstIndex(of: card.condition) ?? 0) },
                set: { newValue in
                    var updated = card
                    let i = Int(newValue.rounded())
                    updated.condition = CardCondition.allCases[min(max(i, 0), CardCondition.allCases.count - 1)]
                    store.update(updated)
                    let gen = UISelectionFeedbackGenerator()
                    gen.selectionChanged()
                }
            )
            Slider(value: index, in: 0...Double(CardCondition.allCases.count - 1), step: 1)
                .tint(Theme.gold)

            HStack {
                ForEach(CardCondition.allCases) { c in
                    Text(c.rawValue)
                        .font(.system(size: 10, weight: c == card.condition ? .heavy : .medium, design: .rounded))
                        .foregroundStyle(c == card.condition ? Theme.gold : Theme.textTertiary)
                        .frame(maxWidth: .infinity)
                }
            }
        }
        .padding(16)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 16))
    }

    private func priceGraph(_ card: Card) -> some View {
        VStack(alignment: .leading, spacing: 10) {
            HStack {
                Text("30-Day Trend")
                    .font(.system(size: 15, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
                Spacer()
                Text(formatPercent(card.monthlyChangePercent))
                    .font(.system(size: 13, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.profitColor(card.monthlyChangePercent))
            }
            Sparkline(
                points: card.priceHistory.map { $0.price },
                lineColor: Theme.profitColor(card.monthlyChangePercent)
            )
            .frame(height: 110)
        }
        .padding(16)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 16))
    }

    @ViewBuilder
    private func actions(_ card: Card) -> some View {
        VStack(spacing: 12) {
            if card.status == .owned || card.status == .forSale {
                Button {
                    showSellSheet = true
                } label: {
                    Label("Sell / Trade", systemImage: "dollarsign.arrow.circlepath")
                        .font(.system(size: 16, weight: .bold, design: .rounded))
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 15)
                        .background(LinearGradient(colors: [Theme.amber, Theme.crimson], startPoint: .leading, endPoint: .trailing))
                        .foregroundStyle(.white)
                        .clipShape(.rect(cornerRadius: 14))
                }

                Button {
                    var updated = card
                    updated.status = card.status == .forSale ? .owned : .forSale
                    store.update(updated)
                } label: {
                    Label(card.status == .forSale ? "Remove from Sale" : "Mark for Sale",
                          systemImage: "tag.fill")
                        .font(.system(size: 15, weight: .semibold, design: .rounded))
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 13)
                        .background(Theme.surface)
                        .foregroundStyle(Theme.textPrimary)
                        .clipShape(.rect(cornerRadius: 14))
                }
            }

            if card.status == .wishlist {
                Button {
                    var updated = card
                    updated.status = .owned
                    updated.pricePaid = card.targetPrice ?? card.marketPrice
                    updated.dateAdded = Date()
                    store.update(updated)
                    dismiss()
                } label: {
                    Label("Move to Portfolio", systemImage: "plus.square.fill.on.square.fill")
                        .font(.system(size: 16, weight: .bold, design: .rounded))
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 15)
                        .background(LinearGradient(colors: [Theme.violet, Theme.sky], startPoint: .leading, endPoint: .trailing))
                        .foregroundStyle(.white)
                        .clipShape(.rect(cornerRadius: 14))
                }
            }

            Button(role: .destructive) {
                store.remove(card)
                dismiss()
            } label: {
                Label("Remove from Collection", systemImage: "trash")
                    .font(.system(size: 14, weight: .semibold, design: .rounded))
                    .frame(maxWidth: .infinity)
                    .padding(.vertical, 12)
                    .foregroundStyle(Theme.crimson)
            }
        }
    }

    // MARK: Small helpers

    private func metaChip(_ text: String, system: String) -> some View {
        HStack(spacing: 4) {
            Image(systemName: system).font(.system(size: 9))
            Text(text).lineLimit(1)
        }
        .font(.system(size: 11, weight: .semibold))
        .foregroundStyle(Theme.textSecondary)
        .padding(.horizontal, 10)
        .padding(.vertical, 6)
        .background(Theme.bgElevated)
        .clipShape(Capsule())
    }

    private func statBox(title: String, value: String, color: Color) -> some View {
        VStack(spacing: 4) {
            Text(title.uppercased())
                .font(.system(size: 10, weight: .bold, design: .rounded))
                .tracking(0.5)
                .foregroundStyle(Theme.textTertiary)
            Text(value)
                .font(.system(size: 22, weight: .heavy, design: .rounded))
                .foregroundStyle(color)
        }
        .frame(maxWidth: .infinity)
        .padding(.vertical, 14)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 14))
    }
}
