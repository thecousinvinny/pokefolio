import Foundation

/// Collection status of a card in the user's vault.
nonisolated enum CardStatus: String, Codable, CaseIterable, Identifiable, Hashable {
    case owned
    case wishlist
    case forSale
    case sold

    var id: String { rawValue }

    var label: String {
        switch self {
        case .owned: return "Owned"
        case .wishlist: return "Wishlist"
        case .forSale: return "For Sale"
        case .sold: return "Sold"
        }
    }
}

/// Physical condition grade for a card. Affects market value via a multiplier.
nonisolated enum CardCondition: String, Codable, CaseIterable, Identifiable, Hashable {
    case nearMint = "NM"
    case lightlyPlayed = "LP"
    case moderatelyPlayed = "MP"
    case heavilyPlayed = "HP"
    case damaged = "DMG"

    var id: String { rawValue }

    var fullName: String {
        switch self {
        case .nearMint: return "Near Mint"
        case .lightlyPlayed: return "Lightly Played"
        case .moderatelyPlayed: return "Moderately Played"
        case .heavilyPlayed: return "Heavily Played"
        case .damaged: return "Damaged"
        }
    }

    /// Fraction of the NM market price this condition is worth.
    var priceMultiplier: Double {
        switch self {
        case .nearMint: return 1.0
        case .lightlyPlayed: return 0.85
        case .moderatelyPlayed: return 0.68
        case .heavilyPlayed: return 0.5
        case .damaged: return 0.32
        }
    }
}

/// Pokémon TCG card type used for filtering and artwork theming.
nonisolated enum CardType: String, Codable, CaseIterable, Identifiable, Hashable {
    case fire = "Fire"
    case water = "Water"
    case grass = "Grass"
    case lightning = "Lightning"
    case psychic = "Psychic"
    case fighting = "Fighting"
    case darkness = "Darkness"
    case metal = "Metal"
    case dragon = "Dragon"
    case colorless = "Colorless"
    case trainer = "Trainer"
    case energy = "Energy"

    var id: String { rawValue }
}

/// Visual rarity of a card.
nonisolated enum CardRarity: String, Codable, CaseIterable, Identifiable, Hashable {
    case common = "Common"
    case uncommon = "Uncommon"
    case rare = "Rare"
    case holoRare = "Holo Rare"
    case ultraRare = "Ultra Rare"
    case specialArt = "Special Art Rare"
    case secretRare = "Secret Rare"

    var id: String { rawValue }
}

/// Holographic foil pattern.
nonisolated enum HoloPattern: String, Codable, CaseIterable, Identifiable, Hashable {
    case none = "None"
    case standard = "Standard"
    case reverse = "Reverse"
    case galaxy = "Galaxy"

    var id: String { rawValue }
}

/// A single dated price observation, used to draw sparklines/graphs.
nonisolated struct PricePoint: Codable, Hashable, Identifiable {
    var id: UUID = UUID()
    var date: Date
    var price: Double
}

/// A Pokémon TCG card. Market price represents the Near Mint TCGPlayer market value.
nonisolated struct Card: Identifiable, Codable, Hashable {
    var id: UUID = UUID()
    var name: String
    var setName: String
    var setNumber: String
    var artist: String
    var type: CardType
    var rarity: CardRarity
    var holo: HoloPattern

    /// Current Near Mint market price (USD).
    var marketPrice: Double
    /// 30-day price history (NM), oldest first.
    var priceHistory: [PricePoint]

    // Collection metadata --------------------------------------------------
    var status: CardStatus = .owned
    var condition: CardCondition = .nearMint
    /// Price the user paid (owned) — nil if unknown / wishlist.
    var pricePaid: Double?
    /// Target buy price for wishlist cards.
    var targetPrice: Double?
    /// Whether price-drop alerts are enabled (wishlist).
    var alertsEnabled: Bool = false
    var dateAdded: Date = Date()

    /// Market value adjusted for this card's condition.
    var conditionAdjustedValue: Double {
        marketPrice * condition.priceMultiplier
    }

    /// 7-day percentage change derived from price history.
    var weeklyChangePercent: Double {
        guard priceHistory.count >= 2 else { return 0 }
        let recent = priceHistory.suffix(8)
        guard let first = recent.first?.price, first > 0, let last = recent.last?.price else { return 0 }
        return (last - first) / first * 100
    }

    /// 30-day percentage change.
    var monthlyChangePercent: Double {
        guard let first = priceHistory.first?.price, first > 0,
              let last = priceHistory.last?.price else { return 0 }
        return (last - first) / first * 100
    }

    /// Profit vs. what the user paid, at the current condition-adjusted value.
    var unrealizedProfit: Double {
        guard let paid = pricePaid else { return 0 }
        return conditionAdjustedValue - paid
    }
}

/// A completed sale or trade, recorded in the ledger.
nonisolated struct SoldEntry: Identifiable, Codable, Hashable {
    var id: UUID = UUID()
    var card: Card
    var dateSold: Date
    var soldPrice: Double
    var fees: Double
    var shipping: Double
    var costBasis: Double

    /// Net profit after fees, shipping, and original cost.
    var netProfit: Double {
        soldPrice - fees - shipping - costBasis
    }
}
