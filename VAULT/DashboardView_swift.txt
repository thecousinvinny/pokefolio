//  Tab 1 — stats, lifetime earnings, the proud favorite card, and market movers.
//

import SwiftUI

struct DashboardView: View {
    @Environment(CollectionStore.self) private var store
    @Binding var selectedTab: Int

    @State private var animatedValue: Double = 0

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.backgroundGradient.ignoresSafeArea()
                ScrollView {
                    VStack(spacing: 24) {
                        portfolioHeader
                        statRow
                        if let fav = store.favoriteCard {
                            favoriteSection(fav)
                        } else {
                            favoriteEmpty
                        }
                        moversSection
                    }
                    .padding(20)
                    .padding(.bottom, 40)
                }
                .navigationDestination(for: UUID.self) { id in
                    CardDetailView(cardID: id)
                }
            }
            .navigationTitle("Dashboard")
            .toolbarBackground(Theme.bg, for: .navigationBar)
        }
        .onAppear {
            withAnimation(.easeOut(duration: 1.0)) { animatedValue = store.totalPortfolioValue }
        }
    }

    private var portfolioHeader: some View {
        VStack(spacing: 6) {
            Text("TOTAL PORTFOLIO VALUE")
                .font(.system(size: 12, weight: .bold, design: .rounded))
                .tracking(1.5)
                .foregroundStyle(Theme.textSecondary)
            Text(formatPrice(animatedValue))
                .font(.system(size: 52, weight: .heavy, design: .rounded))
                .foregroundStyle(
                    LinearGradient(colors: [Theme.gold, Theme.amber], startPoint: .top, endPoint: .bottom)
                )
                .contentTransition(.numericText())
            HStack(spacing: 6) {
                Image(systemName: store.unrealizedProfit >= 0 ? "arrow.up.right" : "arrow.down.right")
                Text("\(formatPrice(store.unrealizedProfit)) unrealized")
            }
            .font(.system(size: 14, weight: .semibold, design: .rounded))
            .foregroundStyle(Theme.profitColor(store.unrealizedProfit))
        }
        .frame(maxWidth: .infinity)
        .padding(.vertical, 28)
        .background(
            RoundedRectangle(cornerRadius: 24)
                .fill(Theme.surface)
                .overlay(
                    RoundedRectangle(cornerRadius: 24)
                        .stroke(Theme.gold.opacity(0.2), lineWidth: 1)
                )
        )
    }

    private var statRow: some View {
        HStack(spacing: 14) {
            miniStat(title: "Lifetime Earned", value: formatPrice(store.lifetimeEarnings),
                     color: Theme.profitColor(store.lifetimeEarnings), icon: "banknote.fill")
            miniStat(title: "Cards Owned", value: "\(store.ownedCards.count)",
                     color: Theme.sky, icon: "rectangle.stack.fill")
        }
    }

    private func favoriteSection(_ card: Card) -> some View {
        VStack(alignment: .leading, spacing: 12) {
            Label("Favorite Card", systemImage: "crown.fill")
                .font(.system(size: 15, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.gold)

            NavigationLink(value: card.id) {
                VStack(spacing: 0) {
                    Color.black
                        .aspectRatio(0.72, contentMode: .fit)
                        .frame(maxWidth: .infinity)
                        .overlay { CardArtwork(card: card).allowsHitTesting(false) }
                        .clipShape(.rect(topLeadingRadius: 20, topTrailingRadius: 20))
                    HStack {
                        VStack(alignment: .leading, spacing: 2) {
                            Text(card.name)
                                .font(.system(size: 20, weight: .heavy, design: .rounded))
                                .foregroundStyle(Theme.textPrimary)
                            Text(card.setName)
                                .font(.system(size: 13, weight: .medium))
                                .foregroundStyle(Theme.textSecondary)
                        }
                        Spacer()
                        Text(formatPrice(card.marketPrice, compact: true))
                            .font(.system(size: 26, weight: .heavy, design: .rounded))
                            .foregroundStyle(Theme.gold)
                    }
                    .padding(16)
                }
                .background(Theme.surface)
                .clipShape(.rect(cornerRadius: 20))
                .overlay(
                    RoundedRectangle(cornerRadius: 20)
                        .stroke(Theme.gold.opacity(0.35), lineWidth: 1.5)
                )
                .shadow(color: Theme.gold.opacity(0.25), radius: 20, y: 10)
            }
            .buttonStyle(.plain)
        }
    }

    private var favoriteEmpty: some View {
        VStack(spacing: 10) {
            Image(systemName: "crown")
                .font(.system(size: 32))
                .foregroundStyle(Theme.gold.opacity(0.5))
            Text("No favorite yet")
                .font(.system(size: 16, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            Text("Long-press a card in your Portfolio or Wishlist to crown your favorite.")
                .font(.system(size: 13))
                .foregroundStyle(Theme.textSecondary)
                .multilineTextAlignment(.center)
        }
        .frame(maxWidth: .infinity)
        .padding(28)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 20))
    }

    private var moversSection: some View {
        VStack(alignment: .leading, spacing: 12) {
            Label("Market Movers · 7d", systemImage: "chart.line.uptrend.xyaxis")
                .font(.system(size: 15, weight: .bold, design: .rounded))
                .foregroundStyle(Theme.emerald)

            if store.marketMovers.isEmpty {
                Text("No gainers in your portfolio this week.")
                    .font(.system(size: 13))
                    .foregroundStyle(Theme.textSecondary)
                    .frame(maxWidth: .infinity, alignment: .leading)
                    .padding(20)
                    .background(Theme.surface)
                    .clipShape(.rect(cornerRadius: 16))
            } else {
                ForEach(store.marketMovers) { card in
                    NavigationLink(value: card.id) {
                        moverRow(card)
                    }
                    .buttonStyle(.plain)
                }
            }
        }
    }

    private func moverRow(_ card: Card) -> some View {
        HStack(spacing: 14) {
            Color.black
                .frame(width: 50, height: 70)
                .overlay { CardArtwork(card: card, showHoloSheen: false).allowsHitTesting(false) }
                .clipShape(.rect(cornerRadius: 8))
            VStack(alignment: .leading, spacing: 3) {
                Text(card.name)
                    .font(.system(size: 15, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
                Text(formatPrice(card.conditionAdjustedValue))
                    .font(.system(size: 13, weight: .medium))
                    .foregroundStyle(Theme.textSecondary)
            }
            Spacer()
            VStack(alignment: .trailing, spacing: 2) {
                HStack(spacing: 2) {
                    Image(systemName: "arrow.up.right").font(.system(size: 11, weight: .bold))
                    Text(String(format: "%.1f%%", card.weeklyChangePercent))
                        .font(.system(size: 16, weight: .heavy, design: .rounded))
                }
                .foregroundStyle(Theme.emerald)
            }
        }
        .padding(12)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 16))
    }

    private func miniStat(title: String, value: String, color: Color, icon: String) -> some View {
        VStack(alignment: .leading, spacing: 8) {
            Image(systemName: icon)
                .font(.system(size: 18, weight: .bold))
                .foregroundStyle(color)
            Text(value)
                .font(.system(size: 24, weight: .heavy, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            Text(title.uppercased())
                .font(.system(size: 10, weight: .bold, design: .rounded))
                .tracking(0.5)
                .foregroundStyle(Theme.textTertiary)
        }
        .frame(maxWidth: .infinity, alignment: .leading)
        .padding(16)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 18))
    }
}
