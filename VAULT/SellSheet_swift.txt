import SwiftUI

/// Records a sale, pre-filled with the card's current condition-adjusted value.
struct SellSheet: View {
    @Environment(CollectionStore.self) private var store
    @Environment(\.dismiss) private var dismiss

    let card: Card

    @State private var soldPrice: Double
    @State private var fees: Double
    @State private var shipping: Double = 4.50

    init(card: Card) {
        self.card = card
        _soldPrice = State(initialValue: (card.conditionAdjustedValue * 100).rounded() / 100)
        _fees = State(initialValue: (card.conditionAdjustedValue * 0.1 * 100).rounded() / 100)
    }

    private var netProfit: Double {
        soldPrice - fees - shipping - (card.pricePaid ?? 0)
    }

    var body: some View {
        NavigationStack {
            ZStack {
                Theme.backgroundGradient.ignoresSafeArea()
                ScrollView {
                    VStack(spacing: 20) {
                        header
                        field(title: "Sold Price", value: $soldPrice, color: Theme.emerald)
                        field(title: "Fees", value: $fees, color: Theme.amber)
                        field(title: "Shipping", value: $shipping, color: Theme.amber)

                        VStack(spacing: 6) {
                            Text("NET PROFIT")
                                .font(.system(size: 11, weight: .bold, design: .rounded))
                                .tracking(1)
                                .foregroundStyle(Theme.textTertiary)
                            Text(formatPrice(netProfit))
                                .font(.system(size: 38, weight: .heavy, design: .rounded))
                                .foregroundStyle(Theme.profitColor(netProfit))
                            if let paid = card.pricePaid {
                                Text("Cost basis \(formatPrice(paid))")
                                    .font(.system(size: 12))
                                    .foregroundStyle(Theme.textSecondary)
                            }
                        }
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 20)
                        .background(Theme.surface)
                        .clipShape(.rect(cornerRadius: 18))

                        Button {
                            store.sell(card, soldPrice: soldPrice, fees: fees, shipping: shipping)
                            UINotificationFeedbackGenerator().notificationOccurred(.success)
                            dismiss()
                        } label: {
                            Text("Confirm Sale")
                                .font(.system(size: 17, weight: .bold, design: .rounded))
                                .frame(maxWidth: .infinity)
                                .padding(.vertical, 16)
                                .background(LinearGradient(colors: [Theme.amber, Theme.crimson], startPoint: .leading, endPoint: .trailing))
                                .foregroundStyle(.white)
                                .clipShape(.rect(cornerRadius: 14))
                        }
                    }
                    .padding(20)
                }
            }
            .navigationTitle("Sell / Trade")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }.foregroundStyle(Theme.textSecondary)
                }
            }
            .toolbarBackground(Theme.bg, for: .navigationBar)
        }
    }

    private var header: some View {
        HStack(spacing: 14) {
            Color.black
                .frame(width: 56, height: 78)
                .overlay { CardArtwork(card: card, showHoloSheen: false).allowsHitTesting(false) }
                .clipShape(.rect(cornerRadius: 8))
            VStack(alignment: .leading, spacing: 4) {
                Text(card.name)
                    .font(.system(size: 17, weight: .bold, design: .rounded))
                    .foregroundStyle(Theme.textPrimary)
                Text("\(card.setName) · \(card.condition.rawValue)")
                    .font(.system(size: 13))
                    .foregroundStyle(Theme.textSecondary)
            }
            Spacer()
        }
    }

    private func field(title: String, value: Binding<Double>, color: Color) -> some View {
        HStack {
            Text(title)
                .font(.system(size: 15, weight: .semibold, design: .rounded))
                .foregroundStyle(Theme.textPrimary)
            Spacer()
            HStack(spacing: 2) {
                Text("$").foregroundStyle(Theme.textSecondary)
                TextField("0", value: value, format: .number.precision(.fractionLength(2)))
                    .keyboardType(.decimalPad)
                    .multilineTextAlignment(.trailing)
                    .frame(width: 90)
                    .foregroundStyle(color)
                    .font(.system(size: 18, weight: .bold, design: .rounded))
            }
        }
        .padding(16)
        .background(Theme.surface)
        .clipShape(.rect(cornerRadius: 14))
    }
}
